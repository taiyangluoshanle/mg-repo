---
title: Navigation Menu
description: 网站导航菜单，支持下拉内容面板。基于 Base UI Navigation Menu 构建。
---

## 安装

```bash
pnpm add @mg/ui
```

## 基础用法

<ComponentPreview>
  <NavigationMenuBasicDemo />
</ComponentPreview>

```tsx
import {
  NavigationMenuRoot, NavigationMenuList, NavigationMenuItem,
  NavigationMenuTrigger, NavigationMenuContent, NavigationMenuLink,
  NavigationMenuPortal, NavigationMenuPositioner, NavigationMenuPopup,
} from "@mg/ui";

export default function Demo() {
  return (
    <NavigationMenuRoot>
      <NavigationMenuList>
        <NavigationMenuItem value="products">
          <NavigationMenuTrigger>产品</NavigationMenuTrigger>
          <NavigationMenuPortal>
            <NavigationMenuPositioner>
              <NavigationMenuPopup>
                <NavigationMenuContent>
                  <NavigationMenuLink href="#">组件库</NavigationMenuLink>
                </NavigationMenuContent>
              </NavigationMenuPopup>
            </NavigationMenuPositioner>
          </NavigationMenuPortal>
        </NavigationMenuItem>
      </NavigationMenuList>
    </NavigationMenuRoot>
  );
}
```

## API

### NavigationMenuRoot

| 属性 | 类型 | 默认值 | 说明 |
|------|------|--------|------|
| `value` | `string` | — | 受控当前激活项 |
| `defaultValue` | `string` | — | 默认激活项 |
| `onValueChange` | `(value: string) => void` | — | 激活项变化回调 |
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | 方向 |
| `delay` | `number` | `200` | 打开延迟（ms） |
| `closeDelay` | `number` | `200` | 关闭延迟（ms） |
